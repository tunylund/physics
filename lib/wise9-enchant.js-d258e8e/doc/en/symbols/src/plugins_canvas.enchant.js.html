 32     var _touchstartFromDom = function(e) {
 33         var game = enchant.Game.instance;
 34         var group;
 35         for (var i = canvasGroupInstances.length - 1; i >= 0; i--) {
 36             group = canvasGroupInstances[i];
 37             if (group.scene != game.currentScene) continue;
 38             var sp = group._touchstartPropagation(e);
 39             if (sp) {
 40                 touchingEntity = sp;
 41                 touchingGroup = group;
 42                 return;
 43             }
 44         }

 81     enchant.Sprite.prototype.cvsRender = function(ctx) {
 82         var img, imgdata, row, frame;
 83         var sx, sy, sw, sh;
 84         if (this._image) {
 85             frame = Math.abs(this._frame) || 0;
 86             img = this._image;
 87             imgdata = img._element;
 88             row = img.width / this._width | 0;
 89             sx = (frame % row) * this._width;
 90             sy = (frame / row | 0) * this._height % img.height;
 91             sy = Math.min(sy, img.height - this._height);
 92             sw = Math.min(img.width - sx, this._width);
 93             sh = Math.min(img.height - sy, this._height);
 94             ctx.drawImage(imgdata, sx, sy, sw, sh, RENDER_OFFSET, RENDER_OFFSET, this._width + RENDER_OFFSET, this._height + RENDER_OFFSET);
 95         }
 96     };

107     var DetectColorManager = enchant.Class.create({
108         initialize: function(reso, max) {
109             this.reference = new Array();
110             this.detectColorNum = 0;
111             this.colorResolution = reso || 16;
112             this.max = max || 1;
113         },
114         attachDetectColor: function(sprite) {
115             this.detectColorNum += 1;
116             this.reference[this.detectColorNum] = sprite;
117             return this._createNewColor();
118         },
119         detachDetectColor: function(sprite) {
120             var i = this.reference.indexOf(sprite);
121             if (i != -1) {
122                 this.reference[i] = null;
123             }
124         },
125         _createNewColor: function() {
126             var n = this.detectColorNum;
127             var C = this.colorResolution;
128             var d = C / this.max;
129             return [
130                 parseInt((n / C / C) % C) / d,
131                 parseInt((n / C) % C) / d,
132                 parseInt(n % C) / d, 1.0
133             ];
134         },
135         _decodeDetectColor: function(color) {
136             var C = this.colorResolution;
137             return ~~(color[0] * C * C * C / 256)
138                 + ~~(color[1] * C * C / 256)
139                 + ~~(color[2] * C / 256);
140         },
141         getSpriteByColor: function(color) {
142             return this.reference[this._decodeDetectColor(color)];
143         }
144     });

146     var nodesWalker = function(pre, post) {
147         pre = pre || function() {
148         };
149         post = post || function() {
150         };
151         var walker = function() {
152             pre.apply(this, arguments);
153             var child;
154             if (this.childNodes) {
155                 for (var i = 0, l = this.childNodes.length; i < l; i++) {
156                     child = this.childNodes[i];
157                     walker.apply(child, arguments);
158                 }
159             }
160             post.apply(this, arguments);
161         };
162         return walker;
163     };

165     var makeTransformMatrix = function(node, dest) {
166         var x = node.x;
167         var y = node.y;
168         var width = node.width || 0;
169         var height = node.height || 0;
170         var rotation = node.rotation || 0;
171         var scaleX = (typeof node.scaleX == 'number') ? node.scaleX : 1;
172         var scaleY = (typeof node.scaleY == 'number') ? node.scaleY : 1;
173         var theta = Math.PI * rotation / 180;
174         var tmpcos = Math.cos(theta);
175         var tmpsin = Math.sin(theta);
176         var w = (typeof node.originX == 'number') ? node.originX : width / 2;
177         var h = (typeof node.originY == 'number') ? node.originY : height / 2;
178         var a = scaleX * tmpcos;
179         var b = scaleX * tmpsin;
180         var c = scaleY * tmpsin;
181         var d = scaleY * tmpcos;
182         dest[0] = scaleX * tmpcos;
183         dest[1] = scaleX * tmpsin;
184         dest[2] = -scaleY * tmpsin;
185         dest[3] = scaleY * tmpcos;
186         dest[4] = (-a * w + c * h + x + w);
187         dest[5] = (-b * w - d * h + y + h);
188     }

190     var dirtyCheck = function(node) {
191         if (node.__dirty
192             || node._cvsCache.x != node.x
193             || node._cvsCache.y != node.y
194             || node._cvsCache.width != node.width
195             || node._cvsCache.height != node.height) {
196             makeTransformMatrix(node, node._cvsCache.matrix);
197             node.__dirty = false;
198             node._cvsCache.x = node.x;
199             node._cvsCache.y = node.y;
200             node._cvsCache.width = node.width;
201             node._cvsCache.height = node.height;
202         }
203     };

205     var alpha = function(ctx, node) {
206         if (node.alphaBlending) {
207             ctx.globalCompositeOperation = node.alphaBlending;
208         } else {
209             ctx.globalCompositeOperation = "source-atob";
210         }
211         ctx.globalAlpha = node.opacity || 1.0;
212     };

214     var transform = function(ctx, node) {
215         dirtyCheck(node);
216         ctx.transform.apply(ctx, node._cvsCache.matrix);
217     };

219     var render = function(ctx, node) {
220         var game = enchant.Game.instance;
221         if (node.backgroundColor) {
222             ctx.fillStyle = node.backgroundColor;
223             ctx.fillRect(RENDER_OFFSET, RENDER_OFFSET, node.width + RENDER_OFFSET, node.height + RENDER_OFFSET);
224         }
225 
226         if (node.cvsRender) {
227             node.cvsRender(ctx);
228         }
229 
230         if (game._debug) {
231             if (node instanceof enchant.Label || node instanceof enchant.Sprite) {
232                 ctx.strokeStyle = '#ff0000';
233             } else {
234                 ctx.strokeStyle = '#0000ff';
235             }
236             ctx.strokeRect(RENDER_OFFSET, RENDER_OFFSET, node.width + RENDER_OFFSET, node.height + RENDER_OFFSET);
237         }
238     };

241         function(ctx) {
242             ctx.save();
243             alpha(ctx, this);
244             transform(ctx, this);
245             render(ctx, this);
246         },
247         function(ctx) {
248             ctx.restore();

253         return '#' + ("00" + Number(parseInt(arr[0])).toString(16)).slice(-2)
254             + ("00" + Number(parseInt(arr[1])).toString(16)).slice(-2)
255             + ("00" + Number(parseInt(arr[2])).toString(16)).slice(-2);

274     var is__dirty = function() {

276             this.__dirty = true;
277         } else {
278             this.__dirty = false;

282     var attachCache = function(colorManager) {
283         if (this._cvsCache) return;
284         this.addEventListener('render', is__dirty);
285         this._cvsCache = {};
286         this._cvsCache.matrix = [];
287         this._cvsCache.detectColor = array2hexrgb(colorManager.attachDetectColor(this));
288     };
289 
290     var detachCache = function(colorManager) {
291         if (!this._cvsCache) return;
292         this.removeEventListener('render', is__dirty);
293         colorManager.detachDetectColor(this);
294         delete this._cvsCache;
295     };

324         this.dispatchEvent(e);

333     enchant.CanvasGroup = enchant.Class.create(enchant.Group, {
334         initialize: function() {
335             var game = enchant.Game.instance;
336             var that = this;
337             enchant.Group.call(this);
338             this._dirty = false;
339             this._rotation = 0;
340             this._cvsCache = {};
341             this._cvsCache.matrix = [ 1, 0, 0, 1, 0, 0];
342             this._cvsCache.detectColor = '#000000';
343             this.width = game.width;
344             this.height = game.height;
345 
346             var sceneEvents = [
347                 enchant.Event.ADDED_TO_SCENE,
348                 enchant.Event.REMOVED_FROM_SCENE
349             ];
350             sceneEvents.forEach(function(event) {
351                 this.addEventListener(event, function(e) {
352                     this.childNodes.forEach(function(child) {
353                         child.scene = this.scene;
354                         child.dispatchEvent(e);
355                     }, this);
356                 });
357             }, this);
358 
359             this._element = document.createElement('canvas');
360             this._element.width = game.width;
361             this._element.height = game.height;
362             this._element.style.position = 'absolute';
363 
364             this._detect = document.createElement('canvas');
365             this._detect.width = game.width;
366             this._detect.height = game.height;
367             this._detect.style.position = 'absolute';
368 
369             this.context = this._element.getContext('2d');
370             this._dctx = this._detect.getContext('2d');
371 
372             this._colorManager = new DetectColorManager(16, 256);
373 
374             if (enchant.ENV.TOUCH_ENABLED) {
375                 this._element.addEventListener('touchstart', function(e) {
376                     var touches = e.touches;
377                     for (var i = 0, len = touches.length; i < len; i++) {
378                         e = new enchant.Event('touchstart');
379                         e.identifier = touches[i].identifier;
380                         e._initPosition(touches[i].pageX, touches[i].pageY);
381                         _touchstartFromDom.call(that, e);
382                     }
383                 }, false);
384                 this._element.addEventListener('touchmove', function(e) {
385                     var touches = e.touches;
386                     for (var i = 0, len = touches.length; i < len; i++) {
387                         e = new enchant.Event('touchmove');
388                         e.identifier = touches[i].identifier;
389                         e._initPosition(touches[i].pageX, touches[i].pageY);
390                         _touchmoveFromDom.call(that, e);
391                     }
392                 }, false);
393                 this._element.addEventListener('touchend', function(e) {
394                     var touches = e.changedTouches;
395                     for (var i = 0, len = touches.length; i < len; i++) {
396                         e = new enchant.Event('touchend');
397                         e.identifier = touches[i].identifier;
398                         e._initPosition(touches[i].pageX, touches[i].pageY);
399                         _touchendFromDom.call(that, e);
400                     }
401                 }, false);
402             } else {
403                 this._element.addEventListener('mousedown', function(e) {
404                     var x = e.pageX;
405                     var y = e.pageY;
406                     e = new enchant.Event('touchstart');
407                     e.identifier = game._mousedownID;
408                     e._initPosition(x, y);
409                     _touchstartFromDom.call(that, e);
410                     that._mousedown = true;
411                 }, false);
412                 game._element.addEventListener('mousemove', function(e) {
413                     if (!that._mousedown) return;
414                     var x = e.pageX;
415                     var y = e.pageY;
416                     e = new enchant.Event('touchmove');
417                     e.identifier = game._mousedownID;
418                     e._initPosition(x, y);
419                     _touchmoveFromDom.call(that, e);
420                 }, false);
421                 game._element.addEventListener('mouseup', function(e) {
422                     if (!that._mousedown) return;
423                     var x = e.pageX;
424                     var y = e.pageY;
425                     e = new enchant.Event('touchend');
426                     e.identifier = game._mousedownID;
427                     e._initPosition(x, y);
428                     _touchendFromDom.call(that, e);
429                     that._mousedown = false;
430                 }, false);
431             }
432             var start = [
433                 enchant.Event.ENTER,
434                 enchant.Event.ADDED_TO_SCENE
435             ];
436             var end = [
437                 enchant.Event.EXIT,
438                 enchant.Event.REMOVED_FROM_SCENE
439             ];
440             start.forEach(function(type) {
441                 this.addEventListener(type, this._startRendering);
442                 this.addEventListener(type, function() {
443                     canvasGroupInstances.push(this);
444                 });
445             }, this);
446             end.forEach(function(type) {
447                 this.addEventListener(type, this._stopRendering);
448                 this.addEventListener(type, function() {
449                     var i = canvasGroupInstances.indexOf(this);
450                     if (i != -1) {
451                         canvasGroupInstances.splice(i, 1);
452                     }
453                 });
454             }, this);
455 
456             this._onexitframe = function() {
457                 var ctx = that.context;
458                 ctx.clearRect(0, 0, game.width, game.height);
459                 checkCache.call(that, that._colorManager);
460                 rendering.call(that, ctx);
461             };
462         },
463         _startRendering: function() {
464             var game = enchant.Game.instance;
465             if (!game._listeners['exitframe']) game._listeners['exitframe'] = [];
466             game._listeners['exitframe'].push(this._onexitframe);
467         },
468         _stopRendering: function() {
469             var game = enchant.Game.instance;
470             game.removeEventListener('exitframe', this._onexitframe);
471         },
472         _getEntityByPosition: function(x, y) {
473             var ctx = this._dctx;
474             ctx.clearRect(0, 0, this.width, this.height);
475             detectrendering.call(this, ctx);
476             var color = ctx.getImageData(x, y, 1, 1).data;
477             return this._colorManager.getSpriteByColor(color);
478         },
479         _touchstartPropagation: function(e) {
480             var sp = this._getEntityByPosition(e.x, e.y);
481             if (sp) {
482                 this._touching = sp;
483                 propagationUp.call(this._touching, e, this.parentNode);
484             } else {
485                 sp = null;
486             }
487             return sp;
488         },
489         _touchmovePropagation: function(e) {
490             if (this._touching != null) {
491                 propagationUp.call(this._touching, e, this.parentNode);
492             }
493         },
494         _touchendPropagation: function(e) {
495             if (this._touching != null) {
496                 propagationUp.call(this._touching, e, this.parentNode);
497                 this._touching = null;
498             }
499         },
500         rotation: {
501             get: function() {
502                 return this._rotation;
503             },
504             set: function(rot) {
505                 this._rotation = rot;
506                 this._dirty = true;
507             }
508         },
509         scaleX: {
510             get: function() {
511                 return this._scaleX;
512             },
513             set: function(scale) {
514                 this._scaleX = scale;
515                 this._dirty = true;
516             }
517         },
518         scaleY: {
519             get: function() {
520                 return this._scaleY;
521             },
522             set: function(scale) {
523                 this._scaleY = scale;
524                 this._dirty = true;
525             }
526         },
527         addChild: function(node) {
528             this.childNodes.push(node);
529             node.parentNode = this;
530             node.dispatchEvent(new enchant.Event('added'));
531             if (this.scene) {
532                 node.scene = this.scene;
533                 var e = new enchant.Event('addedtoscene');
534                 _onaddedtoscene.call(node, e, this._colorManager);
535             }
536         },
537         insertBefore: function(node, reference) {
538             var i = this.childNodes.indexOf(reference);
539             if (i != -1) {
540                 this.childNodes.splice(i, 0, node);
541                 node.dispatchEvent(new enchant.Event('added'));
542                 if (this.scene) {
543                     node.scene = this.scene;
544                     var e = new enchant.Event('addedtoscene');
545                     _onaddedtoscene.call(node, e, this._colorManager);
546                 }
547             } else {
548                 this.addChild(node);
549             }
550         },
551         removeChild: function(node) {
552             var i;
553             if ((i = this.childNodes.indexOf(node)) != -1) {
554                 this.childNodes.splice(i, 1);
555             }
556             node.parentNode = null;
557             node.dispatchEvent(new enchant.Event('removed'));
558             if (this.scene) {
559                 node.scene = null;
560                 var e = new enchant.Event('removedfromscene');
561                 _onremovedfromscene.call(node, e, this._colorManager);
562             }
563         }
564     });
