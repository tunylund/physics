 99 /**
100  * Export library classes globally.
101  *
102  * When no arguments are delivered, all classes defined in enchant.js as well as all classes defined in
103  * plugins will be exported. When more than one argument is delivered, by default only classes defined
104  * in enchant.js will be exported. When you wish to export plugin classes you must explicitly deliver  *  * plugin identifiers as arguments.
105  *
106  * @example
107  *   enchant();     // All classes will be exported.
108  *   enchant('');   // Only classes in enchant.js will be exported.
109  *   enchant('ui'); // enchant.js classes and ui.enchant.js classes will be exported.
110  *
111  * @param {...String} [modules] Export module. Multiple designations possible.
112  */
113 var enchant = function(modules) {
114     if (modules != null) {
115         if (!(modules instanceof Array)) {
116             modules = Array.prototype.slice.call(arguments);
117         }
118         modules = modules.filter(function(module) {
119             return [module].join();
120         });
121     }
122     (function include(module, prefix) {
123         var submodules = [],
124             i, len;
125         for (var prop in module){
126             if (module.hasOwnProperty(prop)) {
127                 if (typeof module[prop] === 'function') {
128                     window[prop] = module[prop];
129                 } else if (typeof module[prop] === 'object' && Object.getPrototypeOf(module[prop]) === Object.prototype) {
130                     if (modules == null) {
131                         submodules.push(prop);
132                     } else {
133                         i = modules.indexOf(prefix + prop);
134                         if (i !== -1) {
135                             submodules.push(prop);
136                             modules.splice(i, 1);
137                         }
138                     }
139                 }
140             }
141         }
142 
143         for (i = 0, len = submodules.length; i < len; i++) {
144             include(module[submodules[i]], prefix + submodules[i] + '.');
145         }
146     }(enchant, ''));
147 
148     if (modules != null && modules.length) {
149         throw new Error('Cannot load module: ' + modules.join(', '));
150     }
151 };

192 /**
193  * Create class.
194  *
195  * When making classes that succeed other classes, the previous class is used as a base with
196  * constructor as default. In order to override the constructor, it is necessary to explicitly
197  * call up the previous constructor to use it.
198  *
199  * @example
200  *   var Ball = Class.create({ // Creates independent class.
201  *       initialize: function(radius) { ... }, // Method definition.
202  *       fall: function() { ... }
203  *   });
204  *
205  *   var Ball = Class.create(Sprite);  // Creates a class succeeding "Sprite"
206  *   var Ball = Class.create(Sprite, { // Creates a class succeeding "Sprite"
207  *       initialize: function(radius) { // Overwrites constructor
208  *          Sprite.call(this, radius*2, radius*2); // Applies previous constructor.
209  *          this.image = game.assets['ball.gif'];
210  *       }
211  *   });
212  *
213  * @param {Function} [superclass] Preceding class.
214  * @param {*} [definition] Class definition.
215  * @static
216  */
217 enchant.Class.create = function(superclass, definition) {
218     if (arguments.length === 0) {
219         return enchant.Class.create(Object, definition);
220     } else if (arguments.length === 1 && typeof arguments[0] !== 'function') {
221         return enchant.Class.create(Object, arguments[0]);
222     }
223 
224     for (var prop in definition){
225         if (definition.hasOwnProperty(prop)) {
226             if (typeof definition[prop] === 'object' && Object.getPrototypeOf(definition[prop]) === Object.prototype) {
227                 if (!('enumerable' in definition[prop])){
228                     definition[prop].enumerable = true;
229                 }
230             } else {
231                 definition[prop] = { value: definition[prop], enumerable: true, writable: true };
232             }
233         }
234     }
235     var Constructor = function() {
236         if (this instanceof Constructor) {
237             Constructor.prototype.initialize.apply(this, arguments);
238         } else {
239             return new Constructor();
240         }
241     };
242     Constructor.prototype = Object.create(superclass.prototype, definition);
243     Constructor.prototype.constructor = Constructor;
244     if (Constructor.prototype.initialize == null) {
245         Constructor.prototype.initialize = function() {
246             superclass.apply(this, arguments);
247         };
248     }
249 
250     return Constructor;
251 };

583 enchant.EventTarget = enchant.Class.create({
584     /**
585      * A class for independent event implementation like DOM Events.
586      * However, does not include phase concept.
587      * @constructs
588      */
589     initialize: function() {
590         this._listeners = {};
591     },
592     /**
593      * Add EventListener.
594      * @param {String} type Event type.
595      * @param {function(e:enchant.Event)} listener EventListener added.
596      */
597     addEventListener: function(type, listener) {
598         var listeners = this._listeners[type];
599         if (listeners == null) {
600             this._listeners[type] = [listener];
601         } else if (listeners.indexOf(listener) === -1) {
602             listeners.unshift(listener);
603 
604         }
605     },
606     /**
607      * Synonym for addEventListener
608      * @param {String} type Event type.
609      * @param {function(e:enchant.Event)} listener EventListener added.
610      */
611     on: function() {
612         this.addEventListener.apply(this, arguments);
613     },
614     /**
615      * Delete EventListener.
616      * @param {String} type Event type.
617      * @param {function(e:enchant.Event)} listener EventListener deleted.
618      */
619     removeEventListener: function(type, listener) {
620         var listeners = this._listeners[type];
621         if (listeners != null) {
622             var i = listeners.indexOf(listener);
623             if (i !== -1) {
624                 listeners.splice(i, 1);
625             }
626         }
627     },
628     /**
629      * Clear EventListener.
630      * @param {String} type Event type.
631      */
632     clearEventListener: function(type) {
633         if (type != null) {
634             delete this._listeners[type];
635         } else {
636             this._listeners = {};
637         }
638     },
639     /**
640      * Issue event.
641      * @param {enchant.Event} e Event issued.
642      */
643     dispatchEvent: function(e) {
644         e.target = this;
645         e.localX = e.x - this._offsetX;
646         e.localY = e.y - this._offsetY;
647         if (this['on' + e.type] != null){
648             this['on' + e.type](e);
649         }
650         var listeners = this._listeners[e.type];
651         if (listeners != null) {
652             listeners = listeners.slice();
653             for (var i = 0, len = listeners.length; i < len; i++) {
654                 listeners[i].call(this, e);
655             }
656         }
657     }
658 });
